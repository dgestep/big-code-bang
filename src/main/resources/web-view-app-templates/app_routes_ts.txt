import {LoginComponent} from './login/login.component'
import {LandingPageComponent} from "./landing.page.component";
import { SessionLostComponent } from "./session-lost/session.lost.component";
import { LoggedInGuard } from "./login/logged-in.guard";
import {UserProfileComponent} from "./user-profile/user.profile.component";
import {UserAddComponent} from "./user-profile/user.add.component";
import {UserAdminComponent} from "./user-profile/user.admin.component";
import {ChangePasswordComponent} from "./user-profile/change.password.component";
import {PasswordResetComponent} from "./login/password.reset.component";
import {CatchResetComponent} from "./login/catch.reset.component";
import {LogoutComponent} from "./login/logout.component"

export const routes = [
    {path: '', redirectTo: '/welcome', pathMatch: 'full' },
    {path: 'login', component: LoginComponent, useAsDefault: true},
    {path: 'session-lost', component: SessionLostComponent},
    {path: 'reset-password', component: PasswordResetComponent},
    {path: 'catch-reset', component: CatchResetComponent},
    {path: 'logout', component: LogoutComponent},
    {path: 'welcome', component: LandingPageComponent, canActivate: [ LoggedInGuard ]},
    {path: 'user-profile/:id', component: UserProfileComponent, canActivate: [ LoggedInGuard ]},
    {path: 'user-add', component: UserAddComponent, canActivate: [ LoggedInGuard ]},
    {path: 'user-admin', component: UserAdminComponent, canActivate: [ LoggedInGuard ]},
    {path: 'change-password', component: ChangePasswordComponent, canActivate: [ LoggedInGuard ]}
];
