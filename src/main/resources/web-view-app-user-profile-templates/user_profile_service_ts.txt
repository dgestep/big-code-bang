import {Injectable} from "@angular/core";
import {UserProfileTabData} from "./user.profile.tab.data";

/*
 * This service is used to communicate with the UserProfile component by telling it which profile to display
 */
@Injectable()
export class UserProfileService {
    private profileList : Array<UserProfileTabData> = new Array<UserProfileTabData>();

    constructor() {}

    public addProfile(uuid : string, tabName : string) : number {
        this.profileList.push(new UserProfileTabData(uuid, tabName));
        return this.profileList.length - 1;
    }

    public getProfile(tabIndex : number) : UserProfileTabData {
        return this.profileList[tabIndex];
    }

    public clearAllProfiles() {
        this.profileList.splice(0,this.profileList.length);
    }
}
